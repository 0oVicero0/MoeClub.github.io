```
{
    "server":"0.0.0.0",
    "server_port":80,
    "local_port":1080,
    "password":"MoeClub.org",
    "timeout":60,
    "method":"chacha20-ietf-poly1305",
    "plugin":"obfs-server",
    "plugin_opts":"obfs=http"
}
```
